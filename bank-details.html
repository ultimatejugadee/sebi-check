(function() { const PREFERRED_LANG_KEY = 'preferredLang'; const currentPath = window.location.pathname.split('/').pop(); // Get the filename (e.g., index.html) const preferredLang = localStorage.getItem(PREFERRED_LANG_KEY); // Define the page where language selection happens const languageSelectionPage = 'index.html'; // If no language preference is set AND the user is NOT on the language selection page, // redirect them to the language selection page. if (!preferredLang && currentPath !== languageSelectionPage) { console.log('No preferred language set. Redirecting to language selection.'); // Use replace to avoid adding the current page to browser history window.location.replace(languageSelectionPage); } else { // Optional: Log status if needed for debugging if (preferredLang) { console.log(`Preferred language '${preferredLang}' is set.`); } if (currentPath === languageSelectionPage) { console.log('Currently on the language selection page.'); } } })(); // Immediately Invoked Function Expression (IIFE) to avoid polluting global scope
